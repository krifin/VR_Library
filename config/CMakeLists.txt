# Automatically go through all sub directories. This allows you to supply your own configuration
# files for CosmoScout VR by putting them into a custom subdirectory here.

FILE(GLOB SUBDIRECTORIES ${CMAKE_CURRENT_SOURCE_DIR}/*)

foreach(SUBDIR ${SUBDIRECTORIES})
  if(EXISTS "${SUBDIR}/CMakeLists.txt")
    add_subdirectory(${SUBDIR})
  endif()
endforeach()
